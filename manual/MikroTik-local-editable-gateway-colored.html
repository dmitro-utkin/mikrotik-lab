MikroTik-local-editable-gateway.rsc

Кольорова версія призначена лише для читання. Для RouterOS використовуйте вихідний файл MikroTik-local-editable-gateway.rsc.

Синій: RouterOS-команди Червоний: значення або блоки для редагування Сірий: пояснення та коментарі
1# =============================================================================
2# MIKROTIK LOCAL EDITABLE GATEWAY TEMPLATE
3# =============================================================================
4# Target: RouterOS 7, clean configuration, new /interface wifi stack.
5# Delivery: local file upload only. No SSH, API, HTTP or HTTPS management.
6#
7# IMPORTANT:
8# 1. Make a backup/export before changing the router.
9# 2. This file expects:
10# /system reset-configuration no-defaults=yes skip-backup=yes
11# Reset is NOT executed by this file.
12# 3. Search for "EDIT:" and review every marked value.
13# 4. Do not store real WiFi passwords in GitHub.
14# 5. Test locally before import:
15# /import file-name=MikroTik-local-editable-gateway.rsc \
16# verbose=yes dry-run=yes
17# 6. Import only while local recovery is available: MAC Winbox or console.
18#
19# CURRENT EXAMPLE TOPOLOGY:
20# ether1 = WAN, DHCP client
21# ether2 = trunk, tagged VLAN 888/100/200
22# ether3 = access VLAN 100 (WORK)
23# ether5 = access VLAN 888 (MGMT)
24# wifi1 = access VLAN 100 (WORK)
25# wifi2 = access VLAN 200 (GUEST)
26#
27# VLAN 888 = MGMT = 10.11.88.0/24
28# VLAN 100 = WORK = 10.11.100.0/24
29# VLAN 200 = GUEST = 10.11.200.0/24
30# =============================================================================

64# SAFETY GUARD
65# Remove these checks only when intentionally converting this file into a
66# migration script for an existing configuration.

79# INTERFACE PREFLIGHT
80# EDIT: change interface names here and everywhere below when hardware differs.
81# Delete WiFi checks if the device has no local WiFi.
82# -----------------------------------------------------------------------------
83:if ([:len [/interface find where name="ether1"]] = 0) do={
84 :error "Missing interface ether1 (WAN)"

102# -----------------------------------------------------------------------------
103# DEVICE IDENTITY
104# EDIT: router name and time zone.
105# -----------------------------------------------------------------------------
106/system identity
107set name=$routerName
108
109/system clock
110set time-zone-name="Europe/Kyiv"

112# -----------------------------------------------------------------------------
113# BRIDGE
114# VLAN filtering stays disabled until the final section to reduce lockout risk.
115# The bridge itself is the CPU port and accepts only tagged VLAN traffic.
116# EDIT: rename bridge-vlan only if all references below are changed too.
117# -----------------------------------------------------------------------------
118/interface bridge
119add name="bridge-vlan" \
120 protocol-mode=rstp \
121 vlan-filtering=no \
122 ingress-filtering=yes \
123 frame-types=admit-only-vlan-tagged \
124 comment="MAIN VLAN BRIDGE"
125
126# -----------------------------------------------------------------------------
127# INTERFACE LISTS
128# WAN is used by firewall/NAT. LAN contains routed VLAN interfaces.
129# MGMT contains only the management VLAN interface.
130# -----------------------------------------------------------------------------
131/interface list
132add name=WAN comment="Internet uplinks"
133add name=LAN comment="Local routed VLANs"
134add name=MGMT comment="Local management only"
135
136/interface list member
137add interface=ether1 list=WAN comment="EDIT: WAN"
138
139# -----------------------------------------------------------------------------
140# L3 VLAN INTERFACES
141# These interfaces terminate VLANs on the router CPU.
142# EDIT: VLAN IDs and names when the task requires different networks.
143# -----------------------------------------------------------------------------
144/interface vlan
145add interface=bridge-vlan name=vlan888-mgmt vlan-id=888 \
146 comment="MGMT VLAN"
147add interface=bridge-vlan name=vlan100-work vlan-id=100 \
148 comment="WORK VLAN"
149add interface=bridge-vlan name=vlan200-guest vlan-id=200 \
150 comment="GUEST VLAN"
151
152/interface list member
153add interface=vlan888-mgmt list=LAN
154add interface=vlan888-mgmt list=MGMT
155add interface=ether5 list=MGMT comment="Local MAC Winbox recovery"
156add interface=vlan100-work list=LAN
157add interface=vlan200-guest list=LAN
158
159# -----------------------------------------------------------------------------
160# WIFI SECURITY AND SSIDS
161# EDIT: country, SSIDs and both passphrases.
162# Never commit a copy containing real passphrases.
163# If the router uses legacy /interface wireless, replace this whole section.
164# -----------------------------------------------------------------------------
165/interface wifi security
166add name=sec-work \
167 authentication-types=wpa2-psk,wpa3-psk \
168 passphrase=$workWifiPassword
169add name=sec-guest \
170 authentication-types=wpa2-psk,wpa3-psk \
171 passphrase=$guestWifiPassword
172
173/interface wifi datapath
174add name=dp-work client-isolation=no
175add name=dp-guest client-isolation=yes
176
177/interface wifi configuration
178add name=cfg-work \
179 mode=ap \
180 ssid=$workSsid \
181 country="Ukraine" \
182 installation=indoor \
183 security=sec-work \
184 datapath=dp-work
185add name=cfg-guest \
186 mode=ap \
187 ssid=$guestSsid \
188 country="Ukraine" \
189 installation=indoor \
190 security=sec-guest \
191 datapath=dp-guest
192
193/interface wifi
194set [find where default-name=wifi1] configuration=cfg-work disabled=no
195set [find where default-name=wifi2] configuration=cfg-guest disabled=no
196
197# -----------------------------------------------------------------------------
198# BRIDGE PORTS
199# Trunk: accepts tagged frames only.
200# Access: accepts untagged frames and assigns the configured PVID.
201# EDIT: add/remove access ports according to the device task.
202# -----------------------------------------------------------------------------
203/interface bridge port
204add bridge=bridge-vlan interface=ether2 \
205 ingress-filtering=yes \
206 frame-types=admit-only-vlan-tagged \
207 comment="TRUNK VLAN 888,100,200"
208
209add bridge=bridge-vlan interface=ether5 pvid=888 \
210 ingress-filtering=yes \
211 frame-types=admit-only-untagged-and-priority-tagged \
212 comment="MGMT ACCESS VLAN 888"
213
214add bridge=bridge-vlan interface=ether3 pvid=100 \
215 ingress-filtering=yes \
216 frame-types=admit-only-untagged-and-priority-tagged \
217 comment="WORK ACCESS VLAN 100"
218
219add bridge=bridge-vlan interface=wifi1 pvid=100 \
220 ingress-filtering=yes \
221 frame-types=admit-only-untagged-and-priority-tagged \
222 comment="WORK WIFI VLAN 100"
223
224add bridge=bridge-vlan interface=wifi2 pvid=200 \
225 ingress-filtering=yes \
226 frame-types=admit-only-untagged-and-priority-tagged \
227 comment="GUEST WIFI VLAN 200"
228
229# -----------------------------------------------------------------------------
230# BRIDGE VLAN TABLE
231# Keep one entry per VLAN when access ports are present.
232# bridge-vlan is tagged because the router CPU terminates all three VLANs.
233# ether2 is tagged because it is the trunk.
234# EDIT: keep this table synchronized with bridge port PVID values.
235# -----------------------------------------------------------------------------
236/interface bridge vlan
237add bridge=bridge-vlan vlan-ids=888 \
238 tagged=bridge-vlan,ether2 \
239 untagged=ether5 \
240 comment="MGMT VLAN"
241
242add bridge=bridge-vlan vlan-ids=100 \
243 tagged=bridge-vlan,ether2 \
244 untagged=ether3,wifi1 \
245 comment="WORK VLAN"
246
247add bridge=bridge-vlan vlan-ids=200 \
248 tagged=bridge-vlan,ether2 \
249 untagged=wifi2 \
250 comment="GUEST VLAN"
251
252# -----------------------------------------------------------------------------
253# ROUTER ADDRESSES
254# EDIT: gateway addresses when subnets change.
255# -----------------------------------------------------------------------------
256/ip address
257add address=10.11.88.1/24 interface=vlan888-mgmt comment="MGMT gateway"
258add address=10.11.100.1/24 interface=vlan100-work comment="WORK gateway"
259add address=10.11.200.1/24 interface=vlan200-guest comment="GUEST gateway"
260
261# -----------------------------------------------------------------------------
262# DHCP
263# EDIT: pools, lease times, gateways and DNS servers.
264# Do not overlap pools with gateways or statically assigned addresses.
265# -----------------------------------------------------------------------------
266/ip pool
267add name=pool-vlan888 ranges=10.11.88.10-10.11.88.50 \
268 comment="MGMT pool"
269add name=pool-vlan100 ranges=10.11.100.10-10.11.100.100 \
270 comment="WORK pool"
271add name=pool-vlan200 ranges=10.11.200.10-10.11.200.100 \
272 comment="GUEST pool"
273
274/ip dhcp-server
275add name=dhcp-vlan888 interface=vlan888-mgmt \
276 address-pool=pool-vlan888 lease-time=1h disabled=no
277add name=dhcp-vlan100 interface=vlan100-work \
278 address-pool=pool-vlan100 lease-time=1h disabled=no
279add name=dhcp-vlan200 interface=vlan200-guest \
280 address-pool=pool-vlan200 lease-time=1h disabled=no
281
282/ip dhcp-server network
283add address=10.11.88.0/24 gateway=10.11.88.1 \
284 dns-server=1.1.1.1,8.8.8.8 comment="MGMT network"
285add address=10.11.100.0/24 gateway=10.11.100.1 \
286 dns-server=1.1.1.1,8.8.8.8 comment="WORK network"
287add address=10.11.200.0/24 gateway=10.11.200.1 \
288 dns-server=1.1.1.1,8.8.8.8 comment="GUEST network"
289
290# -----------------------------------------------------------------------------
291# WAN
292# Current example expects DHCP from the ISP on ether1.
293# EDIT: replace this section for static IP, PPPoE or another WAN method.
294# -----------------------------------------------------------------------------
295/ip dhcp-client
296add interface=ether1 \
297 add-default-route=yes \
298 use-peer-dns=no \
299 disabled=no \
300 comment="WAN DHCP"
301
302/ip dns
303set servers=1.1.1.1,8.8.8.8 allow-remote-requests=no
304
305# -----------------------------------------------------------------------------
306# FIREWALL ADDRESS LISTS
307# LOCAL_SUBNETS is used for NAT policy and inter-VLAN isolation.
308# MGMT_SUBNETS is allowed to manage the router through local Winbox.
309# -----------------------------------------------------------------------------
310/ip firewall address-list
311add list=LOCAL_SUBNETS address=10.11.88.0/24 comment="MGMT"
312add list=LOCAL_SUBNETS address=10.11.100.0/24 comment="WORK"
313add list=LOCAL_SUBNETS address=10.11.200.0/24 comment="GUEST"
314add list=MGMT_SUBNETS address=10.11.88.0/24 comment="MGMT only"
315
316# -----------------------------------------------------------------------------
317# INPUT FIREWALL: traffic to the router itself.
318# SSH/API/Web are disabled later. Local Winbox is allowed only from MGMT.
319# -----------------------------------------------------------------------------
320/ip firewall filter
321add chain=input action=accept \
322 connection-state=established,related,untracked \
323 comment="INPUT established"
324add chain=input action=drop connection-state=invalid \
325 comment="INPUT invalid"
326add chain=input action=accept protocol=icmp \
327 comment="INPUT ICMP"
328add chain=input action=accept protocol=udp src-port=68 dst-port=67 \
329 in-interface-list=LAN \
330 comment="INPUT DHCP"
331add chain=input action=accept src-address-list=MGMT_SUBNETS \
332 comment="INPUT local management"
333add chain=input action=drop \
334 comment="INPUT drop all"
335
336# -----------------------------------------------------------------------------
337# FORWARD FIREWALL: traffic through the router.
338# All local VLANs can reach WAN. Routing between local VLANs is blocked.
339# EDIT: add explicit allow rules BEFORE "FORWARD isolate VLANs" when needed.
340# -----------------------------------------------------------------------------
341add chain=forward action=fasttrack-connection \
342 connection-state=established,related hw-offload=yes \
343 comment="FORWARD fasttrack"
344add chain=forward action=accept \
345 connection-state=established,related,untracked \
346 comment="FORWARD established"
347add chain=forward action=drop connection-state=invalid \
348 comment="FORWARD invalid"
349add chain=forward action=accept \
350 src-address-list=LOCAL_SUBNETS out-interface-list=WAN \
351 comment="FORWARD local to internet"
352add chain=forward action=drop \
353 src-address-list=LOCAL_SUBNETS dst-address-list=LOCAL_SUBNETS \
354 comment="FORWARD isolate VLANs"
355add chain=forward action=drop \
356 connection-state=new in-interface-list=WAN connection-nat-state=!dstnat \
357 comment="FORWARD drop unsolicited WAN"
358add chain=forward action=drop \
359 comment="FORWARD drop all"
360
361# -----------------------------------------------------------------------------
362# NAT
363# Masquerade is appropriate here because the WAN address is received by DHCP.
364# -----------------------------------------------------------------------------
365/ip firewall nat
366add chain=srcnat action=masquerade out-interface-list=WAN \
367 comment="NAT dynamic WAN"
368
369# -----------------------------------------------------------------------------
370# MANAGEMENT SERVICES
371# Policy: no SSH, API, HTTP or HTTPS management.
372# Winbox remains available only from the local MGMT subnet.
373# If network Winbox is also prohibited, set winbox disabled=yes and use console.
374# -----------------------------------------------------------------------------
375/ip service
376set [find where name=telnet] disabled=yes
377set [find where name=ftp] disabled=yes
378set [find where name=www] disabled=yes
379set [find where name=www-ssl] disabled=yes
380set [find where name=ssh] disabled=yes
381set [find where name=api] disabled=yes
382set [find where name=api-ssl] disabled=yes
383set [find where name=winbox] disabled=no address=10.11.88.0/24
384
385# Limit local discovery and MAC Winbox to the management interface list.
386/ip neighbor discovery-settings
387set discover-interface-list=MGMT
388
389/tool mac-server
390set allowed-interface-list=none
391
392/tool mac-server mac-winbox
393set allowed-interface-list=MGMT
394
395/tool bandwidth-server
396set enabled=no
397
398/tool romon
399set enabled=no
400

402# ENABLE VLAN FILTERING LAST
403# This command can disconnect the current session. Local recovery must be ready.
404# -----------------------------------------------------------------------------
405/interface bridge
406set [find where name=bridge-vlan] vlan-filtering=yes
